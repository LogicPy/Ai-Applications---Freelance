# TempMem Framework
class TempMem:
    def __init__(self, storage_file: str = "temp_mem.json"):
        """
        Initialize the TempMem framework for time-aware memory management.
        
        Args:
            storage_file (str): Path to JSON file for persistent storage.
        """
        self.memories: List[Dict[str, Any]] = []
        self.storage_file = storage_file
        self.load_memories()

    def load_memories(self):
        """Load memories from storage file, reconstructing conditions."""
        if os.path.exists(self.storage_file):
            try:
                with open(self.storage_file, 'r') as f:
                    loaded_memories = json.load(f)
                valid_memories = []
                for memory in loaded_memories:
                    # Skip memories without condition_data or with invalid conditions
                    if 'condition_data' not in memory:
                        print(f"Skipping invalid memory {memory.get('id', 'unknown')}: Missing condition_data")
                        continue
                    try:
                        # Convert timestamps to datetime
                        memory['created_at'] = datetime.fromisoformat(memory['created_at'])
                        if memory['expires_at']:
                            memory['expires_at'] = datetime.fromisoformat(memory['expires_at'])
                        # Reconstruct condition
                        memory['condition'] = self._rebuild_condition(memory['condition_data'])
                        valid_memories.append(memory)
                    except Exception as e:
                        print(f"Skipping invalid memory {memory.get('id', 'unknown')}: {e}")
                self.memories = valid_memories
                if not valid_memories:
                    print("No valid memories loaded. Starting fresh.")
                self.save_memories()  # Save cleaned memories
            except Exception as e:
                print(f"Error loading memories: {e}. Starting with empty memory list.")
                self.memories = []

    def _rebuild_condition(self, condition_data: Dict[str, Any]) -> Callable[[], bool]:
        """Rebuild a condition lambda from its serialized data."""
        condition_type = condition_data.get('type')
        if condition_type == 'time_of_day':
            hour = condition_data.get('hour', 0)
            minute = condition_data.get('minute', 0)
            return lambda: datetime.now().hour == hour and datetime.now().minute == minute
        elif condition_type == 'evening':
            return lambda: datetime.now().hour >= 18
        else:
            return lambda: False

    def save_memories(self):
        """Save memories to storage file."""
        try:
            serialized_memories = []
            for memory in self.memories:
                memory_copy = memory.copy()
                memory_copy.pop('condition', None)
                serialized_memories.append(memory_copy)
            with open(self.storage_file, 'w') as f:
                json.dump(serialized_memories, f, default=str)
        except Exception as e:
            print(f"Error saving memories: {e}")

    def remember_if(self, content: Any, condition: Callable[[], bool], condition_data: Dict[str, Any], expires_at: Optional[datetime] = None):
        """
        Store a memory with a condition for recall.
        
        Args:
            content: The memory content (e.g., string, dict, etc.).
            condition: A callable that returns True when the memory should be recalled.
            condition_data: Dictionary describing the condition for serialization.
            expires_at: Optional datetime when the memory should be discarded.
        """
        memory = {
            "content": content,
            "condition": condition,
            "condition_data": condition_data,
            "created_at": datetime.now(pytz.UTC),
            "expires_at": expires_at,
            "id": len(self.memories) + 1
        }
        self.memories.append(memory)
        self.save_memories()
        return memory["id"]

    def recall(self, max_memories: int = 5) -> List[Dict[str, Any]]:
        """
        Cycle through memories and return those whose conditions are met.
        
        Args:
            max_memories: Maximum number of memories to return.
        
        Returns:
            List of memory dictionaries whose conditions are True.
        """
        current_time = datetime.now(pytz.UTC)
        valid_memories = []
        
        for memory in self.memories[:]:
            try:
                if memory["expires_at"] and current_time > memory["expires_at"]:
                    self.memories.remove(memory)
                    continue
                if callable(memory["condition"]) and memory["condition"]():
                    valid_memories.append(memory)
                else:
                    print(f"Skipping memory {memory['id']}: Condition is not callable")
            except Exception as e:
                print(f"Error evaluating condition for memory {memory['id']}: {e}")
        
        valid_memories.sort(key=lambda x: x["created_at"], reverse=True)
        self.save_memories()
        return valid_memories[:max_memories]

    def clear_memory(self, memory_id: int) -> bool:
        """
        Remove a specific memory by ID.
        
        Args:
            memory_id: ID of the memory to remove.
        
        Returns:
            True if memory was removed, False if not found.
        """
        for memory in self.memories[:]:
            if memory["id"] == memory_id:
                self.memories.remove(memory)
                self.save_memories()
                return True
        return False
